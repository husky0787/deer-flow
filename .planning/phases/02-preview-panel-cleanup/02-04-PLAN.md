    ```python
    def _list_newly_uploaded_files(self, thread_id: str, last_message_files: set[str]) -> list[dict]:
        uploads_dir = self._get_uploads_dir(thread_id)

        if not uploads_dir.exists():
            return []

        # 收集目录中所有文件（先扫描一遍）
        all_files = [p for p in uploads_dir.iterdir() if p.is_file()]

        # 构建"系统转换生成的 .md 文件"集合
        # 判断依据：该 .md 文件的 stem 在目录中有对应的 CONVERTIBLE_EXTENSIONS 原始文件
        original_stems = {
            p.stem
            for p in all_files
            if p.suffix.lower() in self._CONVERTIBLE_EXTENSIONS
        }
        converted_md_names = {
            p.name
            for p in all_files
            if p.suffix.lower() == ".md" and p.stem in original_stems
        }

        files = []
        for file_path in sorted(all_files):
            # 跳过已在之前消息中展示过的文件
            if file_path.name in last_message_files:
                continue
            # 跳过系统自动生成的 .md 转换文件
            if file_path.name in converted_md_names:
                continue
            stat = file_path.stat()
            files.append(
                {
                    "filename": file_path.name,
                    "size": stat.st_size,
                    "path": f"/mnt/user-data/uploads/{file_path.name}",
                    "extension": file_path.suffix,
                }
            )

        return files
    ```
